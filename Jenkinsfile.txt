pipeline {
    agent any
    
    stages {
        stage ("checkout from GIT") {
            steps {
                sh 'echo Checkout Correcto' 
            }
        }
        stage ("terraform init") {
            steps {
                dir("/root/tutorial1/") {
                sh 'terraform init'
                }
            }
        }
        stage ("terraform fmt") {
            steps {
                dir("/root/tutorial1/") {
                sh 'terraform fmt'
                }
            }
        }
        stage ("terraform validate") {
            steps {
                dir("/root/tutorial1/") {
                sh 'terraform validate'
                }
            }
        }
        stage ("terrafrom plan") {
            steps {
                dir("/root/tutorial1/") {
                sh 'terraform plan'
                }
            }
        }
        stage ("terraform apply") {
            steps {
                dir("/root/tutorial1/") {
                sh 'terraform apply --auto-approve'
                }
            }
        }
        stage ("Kubernetes Export") {
            steps {
                dir("/root/tutorial1/") {
                sh 'aws eks --region eu-west-3 update-kubeconfig --name demo'
                    }
                }
            }
        stage ("Kubernetes deploy") {
            steps {
                dir("/root/tutorial1/") {
                sh 'kubectl apply -f deployment.yaml'
                sh 'kubectl apply -f public-lb.yaml'
                sh 'kubectl apply -f private-lb.yaml'
                sh 'kubectl apply -f cluster-autoscaler.yaml'
                    }
                }
            }
        }
    }
